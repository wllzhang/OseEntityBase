44struct MapStateInfo {

47 double range; // c: 距离 (range) - 相机到焦点的距离
48 double viewLongitude; // x1: 视角当前经度
49 double viewLatitude; // y1: 视角当前纬度
50 double viewAltitude; // z1: 视角当前高度
51 double mouseLongitude; // x2: 鼠标当前经度
52 double mouseLatitude; // y2: 鼠标当前纬度
53 double mouseAltitude; // z2: 鼠标当前高度
54
55 MapStateInfo()

58 mouseLongitude(116.4), mouseLatitude(39.9), mouseAltitude(MapStateConstants::DEFAULT_ALTITUDE_METERS) {}

62 return std::make_tuple(pitch, heading, range,
